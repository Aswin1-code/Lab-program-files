using namespace std;
class stack{
   int *s;
   int top;
   int size,b;
   public: 
        stack(int sz){
              size=sz;
              s=new int[sz];
              top=-1;
              b=0;
           }
           void push(int x){
              if(top==(size-1)){
                 throw overflow_error("Stack overflow");}
              s[++top]=x;
              b=x;
              cout<<"Pushed: "<<x<<endl;
           }
           int pop(){
              if(b==0){ throw "emptystack";}
              if(top==-1){
                 throw underflow_error("Stack underflow");
                }
              return s[top--];
            }
};
